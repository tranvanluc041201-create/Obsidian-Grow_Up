# C语言复习仪表盘

> [!summary]
> 这页只做一件事：告诉你现在该先复习哪些 C 语言笔记和错题。

> [!note]
> 当前规则：
> - 错题按 `next_review`
> - 笔记按 `review_after`
> - 同一模块里的笔记和错题混合显示
> - 默认同模块内先看错题，再看笔记

```dataviewjs
const today = dv.date("today").startOf("day");

function asArray(value) {
  if (!value) return [];
  if (Array.isArray(value)) return value.map(item => String(item));
  return [String(value)];
}

function inferModule(page) {
  if (page.module) return String(page.module);

  const text = [
    page.file.name,
    page.title ?? "",
    page.file.path,
    ...asArray(page.tags),
    page.source ?? ""
  ].join(" ").toLowerCase();

  if (text.includes("考试技巧") || text.includes("最小公倍数") || text.includes("最大公约数")) return "考试技巧";
  if (
    text.includes("file") ||
    text.includes("fopen") ||
    text.includes("fclose") ||
    text.includes("fread") ||
    text.includes("fwrite") ||
    text.includes("scanf") ||
    text.includes("printf") ||
    text.includes("输入输出") ||
    text.includes("缓冲")
  ) return "输入输出";
  if (
    text.includes("memory") ||
    text.includes("malloc") ||
    text.includes("free") ||
    text.includes("pointer") ||
    text.includes("string") ||
    text.includes("struct") ||
    text.includes("sizeof") ||
    text.includes("alignment") ||
    text.includes("overflow") ||
    text.includes("valgrind") ||
    text.includes("内存") ||
    text.includes("指针") ||
    text.includes("字符串") ||
    text.includes("结构体") ||
    text.includes("二级指针") ||
    text.includes("对齐") ||
    text.includes("越界")
  ) return "内存地址";
  if (
    text.includes("if") ||
    text.includes("switch") ||
    text.includes("while") ||
    text.includes("for") ||
    text.includes("continue") ||
    text.includes("else") ||
    text.includes("逻辑") ||
    text.includes("控制")
  ) return "逻辑控制";
  if (
    text.includes("char") ||
    text.includes("float") ||
    text.includes("double") ||
    text.includes("进制") ||
    text.includes("整数") ||
    text.includes("浮点") ||
    text.includes("数据表示")
  ) return "数据表示";
  if (text.includes("study") || text.includes("学习法")) return "学习方法";
  return "未分类";
}

function displayModule(module) {
  const map = {
    "编译链接": "编译 / 语法 / 宏",
    "内存地址": "内存 / 地址 / 指针",
    "逻辑控制": "逻辑 / 控制流",
    "输入输出": "输入输出 / 缓冲区",
    "数据表示": "数据表示 / 进制 / 浮点",
    "考试技巧": "考试技巧",
    "学习方法": "学习方法",
    "未分类": "未分类"
  };
  return map[module] ?? module;
}

function moduleOrder(module) {
  const order = {
    "编译链接": 1,
    "内存地址": 2,
    "逻辑控制": 3,
    "输入输出": 4,
    "数据表示": 5,
    "考试技巧": 6,
    "学习方法": 7,
    "未分类": 8
  };
  return order[module] ?? 99;
}

function reviewDate(page) {
  if (page.type === "编程错题") {
    return page.next_review ? dv.date(page.next_review).startOf("day") : null;
  }
  if (page.review_after) {
    return dv.date(page.review_after).startOf("day");
  }
  if (page.created) {
    return dv.date(page.created).startOf("day").plus({ days: 1 });
  }
  return null;
}

function displayLink(page) {
  return dv.fileLink(page.file.path, false, page.title ?? page.file.name);
}

function typeLabel(page) {
  return page.type === "编程错题" ? "错题" : "笔记";
}

function typeOrder(page) {
  return page.type === "编程错题" ? 0 : 1;
}

function buildItem(page) {
  const module = inferModule(page);
  const date = reviewDate(page);
  const reasons = [];
  if (page.is_hard === true) reasons.push("高难");
  if (page.repeated === true) reasons.push("重犯");

  return {
    page,
    link: displayLink(page),
    title: page.title ?? page.file.name,
    type: typeLabel(page),
    module,
    moduleLabel: displayModule(module),
    reviewDate: date,
    overdueDays: date ? Math.floor(today.diff(date, "days").days) : 0,
    riskReason: reasons.join(" + "),
    riskWeight: (page.repeated === true ? 2 : 0) + (page.is_hard === true ? 1 : 0)
  };
}

function groupByModule(items) {
  const modules = [...new Set(items.map(item => item.module))];
  return modules
    .sort((a, b) => moduleOrder(a) - moduleOrder(b))
    .map(module => ({
      module,
      label: displayModule(module),
      rows: items.filter(item => item.module === module)
    }));
}

function renderGroupedTable(title, items, columns, rowMapper, emptyText) {
  dv.header(2, title);
  if (items.length === 0) {
    dv.paragraph(emptyText);
    return;
  }

  for (const group of groupByModule(items)) {
    dv.header(3, `${group.label}（${group.rows.length}）`);
    dv.table(columns, group.rows.map(rowMapper));
  }
}

const pages = dv.pages('"10_编程"')
  .where(page =>
    (page.type === "编程错题" || page.type === "编程学习笔记") &&
    page.language === "C"
  )
  .array()
  .map(buildItem)
  .filter(item => item.reviewDate);

const dueToday = pages
  .filter(item => item.reviewDate.toISODate() === today.toISODate())
  .sort((a, b) =>
    moduleOrder(a.module) - moduleOrder(b.module) ||
    typeOrder(a.page) - typeOrder(b.page) ||
    a.title.localeCompare(b.title, "zh-CN")
  );

const overdue = pages
  .filter(item => item.reviewDate < today)
  .sort((a, b) =>
    moduleOrder(a.module) - moduleOrder(b.module) ||
    b.overdueDays - a.overdueDays ||
    typeOrder(a.page) - typeOrder(b.page) ||
    a.title.localeCompare(b.title, "zh-CN")
  );

const highRisk = dv.pages('"10_编程/错题本"')
  .where(page =>
    page.type === "编程错题" &&
    page.language === "C" &&
    (page.is_hard === true || page.repeated === true)
  )
  .array()
  .map(buildItem)
  .sort((a, b) =>
    moduleOrder(a.module) - moduleOrder(b.module) ||
    b.riskWeight - a.riskWeight ||
    a.title.localeCompare(b.title, "zh-CN")
  );

dv.header(2, "总览");
dv.table(["面板", "数量"], [
  ["今天必须复习", dueToday.length],
  ["已逾期", overdue.length],
  ["高危项", highRisk.length]
]);

renderGroupedTable(
  "今天必须复习",
  dueToday,
  ["条目", "类型", "复习日期"],
  item => [item.link, item.type, item.reviewDate.toFormat("yyyy-MM-dd")],
  "今天没有到期的 C 语言笔记或错题。"
);

renderGroupedTable(
  "已逾期",
  overdue,
  ["条目", "类型", "复习日期", "逾期"],
  item => [item.link, item.type, item.reviewDate.toFormat("yyyy-MM-dd"), `${item.overdueDays} 天`],
  "当前没有逾期项。"
);

renderGroupedTable(
  "高危项",
  highRisk,
  ["条目", "危险原因", "下次复习"],
  item => [item.link, item.riskReason, item.reviewDate ? item.reviewDate.toFormat("yyyy-MM-dd") : ""],
  "当前没有标记为高难或重犯的错题。"
);
```
