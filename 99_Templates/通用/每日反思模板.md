## 昨日未完成事项 / 今日承接
<%*
const folder = "20_心理/每日反思";
const yesterday = tp.date.now("YYYY-MM-DD", -1);
const previousPath = `${folder}/${yesterday}.md`;
const previousFile = app.vault.getAbstractFileByPath(previousPath);
let carryovers = [];

if (previousFile) {
  const content = await app.vault.read(previousFile);
  const focusMatch = content.match(/## 明日聚焦([\s\S]*?)(?:\n## |\n---|$)/);
  if (focusMatch) {
    const lines = focusMatch[1]
      .split("\n")
      .map(line => line.trim())
      .filter(line => line.startsWith("- "));

    for (const line of lines) {
      const value = line.replace(/^- [^：:]+[：:]\s*/, "").trim();
      if (value && value !== "___" && value !== "-" && value !== "无") {
        carryovers.push(`- ${value}`);
      }
    }
  }
}

if (carryovers.length === 0) {
  carryovers = ["- 无明确承接项"];
}

tR += carryovers.join("\n");
%>
